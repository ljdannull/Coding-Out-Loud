color_scale = function(x) {
  pct = x / 1560
  return(rgb(pct, max(0, pct - 0.5), (1 - pct) / 2))
}
color_scale_rgb = function(x) {
  pct = x / 1560
  return(c(pct, max(0, pct - 0.5), (1 - pct) / 2))
}
alpha = function(x) {
  pct = x / 1560
  #return(0.2 + pct * 0.3)
  return(0.7 - pct * 0.3)
}
